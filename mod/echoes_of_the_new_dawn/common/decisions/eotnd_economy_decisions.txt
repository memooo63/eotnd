decision_categories = {
    eotnd_economy_category = {
        icon = generic_production
        decisions = {
            eotnd_increase_taxes = {
                cost = 0
                available = { check_variable = { tax_rate < 3 } }
                visible = { always = yes }
                complete_effect = { add_to_variable = { tax_rate = 1 } }
            }

            eotnd_decrease_taxes = {
                cost = 0
                available = { check_variable = { tax_rate > 0 } }
                visible = { always = yes }
                complete_effect = { add_to_variable = { tax_rate = -1 } }
            }

            eotnd_give_aid = {
                cost = 0
                available = { check_variable = { money >= 100 } }
                visible = { always = yes }
                target_trigger = { is_country = yes }
                complete_effect = {
                    if = {
                        limit = { target = { is_major = yes } }
                        subtract_money = { value = 200 }
                        add_to_variable = { influence = 5 }
                        target = {
                            add_money = { value = 200 }
                            add_opinion_modifier = { target = ROOT modifier = eotnd_influence_opinion days = 90 }
                        }
                    }
                    else = {
                        subtract_money = { value = 100 }
                        add_to_variable = { influence = 10 }
                        target = {
                            add_money = { value = 100 }
                            add_opinion_modifier = { target = ROOT modifier = eotnd_influence_opinion days = 90 }
                        }
                    }
                }
            }
        }
    }
}
