eotnd_initialize_economy = {
    set_variable = { money = 1000 }
    set_variable = { debt = 0 }
    set_variable = { influence = 0 }
    set_variable = { tax_rate = 1 }
    set_variable = { income = 0 }
    set_variable = { expenses = 0 }
}

eotnd_set_monarchism = {
    set_politics = { ruling_party = monarchism elections_allowed = no }
    party_popularity = { monarchism = 0.6 }
    create_country_leader = {
        name = "Generic Monarch"
        ideology = monarchism
        traits = { }
        picture = "gfx/leaders/_generic.dds"
        desc = ""
    }
}

eotnd_set_militarist_collectivism = {
    set_politics = { ruling_party = militarist_collectivism elections_allowed = no }
    party_popularity = { militarist_collectivism = 0.6 }
    create_country_leader = {
        name = "Military Council"
        ideology = militarist_collectivism
        traits = { }
        picture = "gfx/leaders/_generic.dds"
        desc = ""
    }
}

eotnd_monthly_economy_tick = {
    set_variable = { expenses = 0 }
    apply_tax_income_monthly = yes
    military_upkeep_monthly = yes
    garrison_upkeep_monthly = yes
    pay_interest_monthly = yes
    clamp_money = yes
    add_to_variable = { influence = -1 }
    if = { limit = { check_variable = { influence < 0 } } set_variable = { influence = 0 } }
}
